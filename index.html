Bounce Game

from js import document, window # Access the canvas canvas = document.getElementById("gameCanvas") ctx = canvas.getContext("2d") # Ball properties ball_x = 250 # Initial x position ball_y = 200 # Initial y position ball_dx = 2 # Speed in x direction ball_dy = 3 # Speed in y direction ball_radius = 10 # Animation loop def update(*args): global ball_x, ball_y, ball_dx, ball_dy # Clear the canvas ctx.clearRect(0, 0, canvas.width, canvas.height) # Draw the ball ctx.beginPath() ctx.arc(ball_x, ball_y, ball_radius, 0, 2 * 3.14159) ctx.fillStyle = "orange" ctx.fill() ctx.closePath() # Update ball position ball_x += ball_dx ball_y += ball_dy # Bounce the ball on canvas edges if ball_x + ball_radius > canvas.width or ball_x - ball_radius < 0: ball_dx = -ball_dx # Reverse x direction if ball_y + ball_radius > canvas.height or ball_y - ball_radius < 0: ball_dy = -ball_dy # Reverse y direction # Request the next frame window.requestAnimationFrame(update) # Start the animation window.requestAnimationFrame(update)
